cmake_minimum_required(VERSION 3.24)
project(HighlineRidge CXX)

# Set C++ standard
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

set(SOURCES
    src/main.cpp
    src/Location.cpp
    src/Button.cpp
    src/ButtonMgr.cpp
    src/RoomLoader.cpp
    src/ConversationManager.cpp
    src/InventoryMgr.cpp
)

# 1. Look for raylib installed via brew/system
find_package(raylib QUIET)

# 2. If not found, download it via FetchContent
if (NOT raylib_FOUND)
    message(STATUS "raylib not found locally → downloading v5.5 with FetchContent...")
    
    include(FetchContent)
    
    FetchContent_Declare(
        raylib
        URL https://github.com/raysan5/raylib/archive/refs/tags/5.5.tar.gz
        DOWNLOAD_EXTRACT_TIMESTAMP TRUE
        # GIT_REPOSITORY https://github.com/raysan5/raylib.git
        # GIT_TAG        5.5
    )

    FetchContent_MakeAvailable(raylib)
endif()

include(FetchContent)
FetchContent_Declare(
    json
    URL https://github.com/nlohmann/json/releases/download/v3.11.3/json.tar.xz
    DOWNLOAD_EXTRACT_TIMESTAMP TRUE
)
FetchContent_MakeAvailable(json)

# 3. Create the executable (use ${PROJECT_NAME} so everything stays consistent)
add_executable(${PROJECT_NAME} ${SOURCES})
set_target_properties(${PROJECT_NAME} PROPERTIES OUTPUT_NAME "Highline Ridge")

# 4. Include your own headers (src/ folder)
target_include_directories(${PROJECT_NAME} PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/src)

# 5. Link raylib
target_link_libraries(${PROJECT_NAME} PRIVATE raylib nlohmann_json::nlohmann_json)

# 6. macOS frameworks (required by raylib on Apple platforms)
if (APPLE)
    target_link_libraries(${PROJECT_NAME} PRIVATE
        "-framework Cocoa"
        "-framework OpenGL"
        "-framework IOKit"
        "-framework CoreVideo"
    )
endif()

# 7. Keep resources beside the executable so texture loads work when launched from build/
add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        "${CMAKE_SOURCE_DIR}/resources"
        "$<TARGET_FILE_DIR:${PROJECT_NAME}>/resources"
    COMMENT "Copying resources to build directory"
)
